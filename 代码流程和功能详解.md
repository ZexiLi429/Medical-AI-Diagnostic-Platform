```javascript
// 伪代码：计算肿瘤体积
function calculateTumorVolume(measurements) {
  let totalVolume = 0;
  
  measurements.forEach(slice => {
    const area = slice.majorAxis * slice.minorAxis; // mm²
    const thickness = slice.sliceThickness; // mm
    const sliceVolume = area * thickness; // mm³
    totalVolume += sliceVolume;
  });
  
  console.log(`肿瘤总体积: ${totalVolume} mm³ = ${totalVolume/1000} cm³`);
  return totalVolume;
}
```

```typescript
async function convertSAMMaskTo3DSegmentation(maskData, viewportId) {
  const { segmentationService } = servicesManager.services;
  
  // 1. 创建新的分割对象

  const volumeId = segmentationService.getSegmentation(segmentationId).representationData.LABELMAP.volumeId;

  });
  
  // 4. 在所有viewport中显示
  await segmentationService.addSegmentationRepresentationToViewport(
    viewportId,
    [{ segmentationId, type: 'Labelmap' }]
  );
  
  // 5. 触发3D渲染
  renderingEngine.render();
  
  console.log('SAM分割已转换为3D模型并显示');
```

#### 方案B: VTK.js直接3D渲染

```javascript
// 更高性能的3D可视化方案

import vtkImageData from '@kitware/vtk.js/Common/DataModel/ImageData';
import vtkVolume from '@kitware/vtk.js/Rendering/Core/Volume';
import vtkVolumeMapper from '@kitware/vtk.js/Rendering/Core/VolumeMapper';

function renderSAMMaskIn3D(maskArray, viewport3D) {
  // 1. 创建VTK ImageData
  const imageData = vtkImageData.newInstance();
  imageData.setDimensions(512, 512, 1);  // 单层
  imageData.setSpacing(0.5, 0.5, 2.5);   // 像素间距
  imageData.getPointData().setScalars(
    vtkDataArray.newInstance({ values: maskArray })
  );
  
  // 2. 创建Volume渲染器
  const actor = vtkVolume.newInstance();
  const mapper = vtkVolumeMapper.newInstance();
  mapper.setInputData(imageData);
  actor.setMapper(mapper);
  
  // 3. 设置颜色映射(红色半透明)
  const ctfun = vtkColorTransferFunction.newInstance();
  ctfun.addRGBPoint(0, 0.0, 0.0, 0.0);   // 背景: 透明
  ctfun.addRGBPoint(1, 1.0, 0.0, 0.0);   // 分割: 红色
  
  const ofun = vtkPiecewiseFunction.newInstance();
  ofun.addPoint(0, 0.0);    // 背景: 完全透明
  ofun.addPoint(1, 0.6);    // 分割: 60%不透明
  
  actor.getProperty().setRGBTransferFunction(0, ctfun);
  actor.getProperty().setScalarOpacity(0, ofun);
  
  // 4. 添加到3D视窗
  viewport3D.getRenderer().addActor(actor);
  viewport3D.render();
}
```

## 🚀 3D显示优化方案（低端设备适配）

### 问题分析

**当前3D渲染存在的问题**:

1. **内存占用过高**: 加载512x512x400的CT需要约400MB显存
2. **WebGL崩溃**: 低端GPU无法处理复杂的Volume Rendering
3. **加载缓慢**: VTK.js初始化耗时3-8秒
4. **帧率过低**: <10 FPS导致交互卡顿

**代码位置**:
```typescript

// 根据设备性能自动调整渲染质量

function detectDeviceCapability() {
  const canvas = document.createElement('canvas');
  const gl = canvas.getContext('webgl2');
  
  if (!gl) {
    return 'low';  // 不支持WebGL2
  }
  
  const debugInfo = gl.getExtension('WEBGL_debug_renderer_info');
  const renderer = gl.getParameter(debugInfo.UNMASKED_RENDERER_WEBGL);
  
  // 检测GPU型号
  if (renderer.includes('Intel HD') || renderer.includes('UHD 620')) {
    return 'low';   // 低端集成显卡
  } else if (renderer.includes('GTX') || renderer.includes('RTX')) {
    return 'high';  // 高端独立显卡
  } else {
    return 'medium';  // 中端显卡
  }
}

function apply3DRenderingConfig(capability) {
  const configs = {
    low: {
      volumeRenderingPreset: 'CT-Bone',  // 简单预设
      blendMode: 'MAXIMUM_INTENSITY_PROJECTION',  // 性能最好
      preferSizeOverAccuracy: true,  // 牺牲精度保证流畅
      gpuMemoryMB: 128,  // 限制显存使用
      targetFrameRate: 15,  // 降低帧率要求
    },
    medium: {
      volumeRenderingPreset: 'CT-Chest-Contrast-Enhanced',
      blendMode: 'AVERAGE_INTENSITY_PROJECTION',
      preferSizeOverAccuracy: false,
      gpuMemoryMB: 512,
      targetFrameRate: 30,
    },
    high: {
      volumeRenderingPreset: 'Custom',
      blendMode: 'COMPOSITE_AVERAGE',  // 最高质量
      preferSizeOverAccuracy: false,
      gpuMemoryMB: 2048,
      targetFrameRate: 60,
    }
  };
  
  return configs[capability];
}
```

### 解决方案5: LOD (Level of Detail) 分层渲染

```typescript
// 根据相机距离动态调整细节层次

class AdaptiveLODController {
  constructor(viewport) {
    this.viewport = viewport;
    this.lodLevels = [
      { distance: 0, resolution: 1.0 },    // 近距离全分辨率
      { distance: 500, resolution: 0.5 },  // 中距离半分辨率
      { distance: 1000, resolution: 0.25 } // 远距离1/4分辨率
    ];
  }
  
  update(cameraDistance) {
    const lod = this.lodLevels.find(l => cameraDistance >= l.distance);
    const currentResolution = lod.resolution;
    
    // 更新渲染分辨率
    this.viewport.setRenderingResolution(currentResolution);
    
    console.log(`相机距离: ${cameraDistance}, LOD: ${currentResolution}x`);
  }
}

// 监听相机变化
viewport.element.addEventListener('camera-modified', (e) => {
  const distance = e.detail.camera.position.distanceTo(target);
  lodController.update(distance);
});
```
